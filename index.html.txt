function generate() {
    const prompt = document.getElementById("prompt").value;
    const output = document.getElementById("output");

    if (!prompt.trim()) {
        output.innerHTML = "<p style='color:red'>プロンプトを入力してください。</p>";
        return;
    }

    // 本当のAPI接続部分はここで書く
    output.innerHTML = `
        <p>生成中…（デモ表示）</p>
        <video src="sample.mp4" autoplay loop muted></video>
        <p style="font-size:12px; opacity:0.7;">※AI連携デモ表示 — 実際にはAPI接続で映像を取得します</p>

}
